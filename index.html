Q5 -Take a variable in outer function and create an inner function to increase the counter every time it is called? 


    ANS:function outerFunction() {
        let counter = 0; // Variable in the outer function
      
        function innerFunction() {
          counter++;
          console.log("Counter: " + counter);
        }
      
        return innerFunction; // Returning the inner function
      }
      
      // Create an instance of the inner function
      const counterFunc = outerFunction();
      
      // Call the inner function multiple times
      counterFunc(); // Output: Counter: 1
      counterFunc(); // Output: Counter: 2
      counterFunc(); //output : counter 3
